struct Question {
    string text;
    string answers[4];
    int correct;
};

int grade12(int correct, int total) {
    int g = round((double)correct / total * 12);
    if (g < 1) g = 1;
    if (g > 12) g = 12;
    return g;
}

void adminAddQuestion() {
    ofstream f("tests.txt", ios::app);

    cin.ignore(); // очищаем перевод строки

    string category, test;
    cout << "Категория: ";
    getline(cin, category);
    cout << "Тест: ";
    getline(cin, test);

    f << "#CATEGORY\n" << category << endl;
    f << "#TEST\n" << test << endl;

    Question q;
    cout << "Текст вопроса: ";
    getline(cin, q.text);

    for (int i = 0; i < 4; i++) {
        cout << "Ответ " << i + 1 << ": ";
        getline(cin, q.answers[i]);
    }

    cout << "Правильный ответ (1-4): ";
    cin >> q.correct;
    q.correct--;

    f << "#QUESTION\n" << q.text << endl;
    for (int i = 0; i < 4; i++)
        f << q.answers[i] << endl;
    f << q.correct << endl;
    f << "#ENDTEST\n";

    cout << "Вопрос добавлен!\n";
}

void passTest() {
    ifstream f("tests.txt");
    vector<string> tests, categories;
    string line, lastCat;

    while (getline(f, line)) {
        if (line == "#CATEGORY")
            getline(f, lastCat);
        else if (line == "#TEST") {
            getline(f, line);
            tests.push_back(line);
            categories.push_back(lastCat);
        }
    }

    if (tests.empty()) {
        cout << "Нет тестов!\n";
        return;
    }

    cout << "\nВыберите тест:\n";
    for (int i = 0; i < tests.size(); i++)
        cout << i + 1 << ". " << categories[i] << " → " << tests[i] << endl;

    int t;
    cin >> t;
    if (t < 1 || t > tests.size()) {
        cout << "Неверный выбор\n";
        return;
    }
    t--;

    f.clear();
    f.seekg(0);

    int correct = 0, total = 0;
    bool active = false;

    while (getline(f, line)) {
        if (line == "#TEST") {
            getline(f, line);
            active = (line == tests[t]);
        }
        else if (active && line == "#QUESTION") {
            Question q;
            total++;

            getline(f, q.text);
            cout << "\n" << q.text << endl;

            for (int i = 0; i < 4; i++) {
                getline(f, q.answers[i]);
                cout << i + 1 << ". " << q.answers[i] << endl;
            }

            f >> q.correct;
            f.ignore(); // убрать \n

            int ans;
            cin >> ans;
            if (ans - 1 == q.correct)
                correct++;
        }
        else if (line == "#ENDTEST" && active)
            break;
    }

    int grade = grade12(correct, total);
    cout << "\nРезультат: " << correct << "/" << total
         << "  Оценка: " << grade << endl;
}

int main() {
    setlocale(LC_ALL, "ru");
    initTests();

    while (true) {
        cout << "\n1. Администратор\n2. Тестируемый\n0. Выход\n";
        int c;
        cin >> c;

        if (c == 0) break;

        if (c == 1) {
            cout << "1. Добавить вопрос\n";
            cin >> c;
            if (c == 1) adminAddQuestion();
        }
        else if (c == 2) {
            cout << "1. Пройти тест\n";
            cin >> c;
            if (c == 1) passTest();
        }
    }

    return 0;
}
